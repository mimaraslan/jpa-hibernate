CREATE DATABASE IF NOT EXISTS dbusiness DEFAULT CHARACTER SET utf8 ;

USE dbusiness;

DROP TABLE dbusiness.customer_detail; 
DROP TABLE dbusiness.customer;

CREATE TABLE CUSTOMER(
CUSTOMER_ID INT( 10 ) NOT NULL AUTO_INCREMENT ,
TITLE VARCHAR( 10 ) NOT NULL ,
NAME VARCHAR( 20 ) NOT NULL ,
PRIMARY KEY ( CUSTOMER_ID )
);
 
CREATE TABLE ORDERS(
ORDER_ID INT( 10 ) NOT NULL AUTO_INCREMENT ,
CODE VARCHAR( 20 ) NOT NULL,
AMOUNT FLOAT(10,2) DEFAULT NULL ,
ORDER_DATE DATE ,
CUSTOMER_ID INT( 10 ) NOT NULL ,
PRIMARY KEY ( ORDER_ID ),
UNIQUE KEY UNI_CO (CODE),
CONSTRAINT FK_CUST FOREIGN KEY (CUSTOMER_ID) REFERENCES CUSTOMER (CUSTOMER_ID)
);
